#### 3. 实现 JavaScript 逻辑

**修改文件**：`templates/index.html` (第 535-635 行)

**核心函数**：

**打开抽屉**：
```javascript
function showCoCreateOptions() {
    openCoCreateDrawer();
}

function openCoCreateDrawer() {
    document.getElementById('coCreateOverlay').classList.add('active');
    document.getElementById('coCreateDrawer').classList.add('active');
}
```

**选择类型**：
```javascript
function selectCoCreateType(type) {
    currentCoCreateType = type;
    
    // 显示输入框
    const inputGroup = document.getElementById('coCreateInput');
    inputGroup.style.display = 'block';
    
    // 滚动到输入框
    setTimeout(() => {
        inputGroup.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    }, 100);
}
```

**提交作品**：
```javascript
async function submitCoCreate() {
    const userInput = document.getElementById('userInput').value.trim();
    
    if (!userInput) {
        alert('请填写你的创作内容');
        return;
    }
    
    if (currentCoCreateType === 'relay') {
        // 接力创作
        const response = await fetch('/api/relay', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({
                original: currentContent || {},
                user_input: userInput
            })
        });
        
        const data = await response.json();
        
        if (data.success || data.type === 'relay') {
            alert('✅ 接力作品提交成功！\n\nAI 续写：' + data.ai_continuation);
            closeCoCreateDrawer();
        }
    } else if (currentCoCreateType === 'challenge') {
        // 夺擂创作
        const response = await fetch('/api/challenge', {
            method: 'POST',
            body: JSON.stringify({
                requirement: {
                    theme: userInput,
                    type: currentContent?.type || 'novel'
                }
            })
        });
        
        const data = await response.json();
        
        if (data.success || data.challenge_mode) {
            alert('✅ 夺擂作品提交成功！\n\n正在生成新作品...');
            closeCoCreateDrawer();
        }
    }
}
```
