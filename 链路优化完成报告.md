#### 节点 2: `/welcome` → `/personality_customization`

**实现方式**：前端 JavaScript + API 调用

**代码位置**：`templates/welcome_premium.html` (第 356-412 行)

```javascript
async function handleContinue() {
    const userName = document.getElementById('userName').value;
    const aiName = document.getElementById('aiName').value;
    
    // 1. 验证
    if (!userName || !aiName) {
        showToast('请填写名字和 AI 分身名字');
        return;
    }
    
    // 2. 保存到 localStorage
    localStorage.setItem('goIn_userName', userName);
    localStorage.setItem('goIn_aiName', aiName);
    
    // 3. 按钮加载状态
    const btn = document.querySelector('.continue-btn');
    btn.classList.add('loading');
    btn.disabled = true;
    
    // 4. API 调用后端
    const response = await fetch('/api/save_avatar_info', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            user_name: userName,
            ai_name: aiName,
            avatar_style: 'human',
            onboarding_completed: true
        })
    });
    
    // 5. 处理响应
    const data = await response.json();
    
    if (data.success) {
        // 6. 显示成功动画
        showSuccessAndRedirect();
    } else {
        throw new Error(data.message || '保存失败');
    }
}

// 成功动画并跳转
function showSuccessAndRedirect() {
    const card = document.querySelector('.welcome-card');
    
    card.innerHTML = `
        <div class="success-checkmark">✓</div>
        <div class="welcome-title" style="text-align: center;">准备好了吗？</div>
        <div class="welcome-desc" style="text-align: center;">
            即将进入你的 AI 世界...
        </div>
    `;
    
    // 延迟跳转（让用户看到动画）
    setTimeout(() => {
        window.location.href = '/personality_customization';
    }, 1500);
}
```

**关键特性**：
- ✅ 表单验证
- ✅ 双重保存（localStorage + 后端 API）
- ✅ 加载状态提示
- ✅ 错误处理
- ✅ 成功动画（对勾标记）
- ✅ 延迟跳转（1.5 秒，配合动画）

### 2. 缓动曲线（Easing Functions）

**标准缓动**（类 iOS）：
```css
--ease-standard: cubic-bezier(0.4, 0.0, 0.2, 1);
```

**减速缓动**（进入动画）：
```css
--ease-decelerate: cubic-bezier(0.0, 0.0, 0.2, 1);
```

**弹跳缓动**（成功动画）：
```css
--ease-bounce: cubic-bezier(0.68, -0.55, 0.265, 1.55);
```
